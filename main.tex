\begin{enumerate}
  \item {[10 pts]} Determine if each of the following
    differentials is exact:

    \begin{enumerate}[(a)]
      \item $dz = ydx - xdy$

        \boxedanswersmall{
          A differential is exact if given:

          \begin{align*}
            Z &= Z(X,Y) \tag{4.16} \\
            dZ &= M dX + N dY \tag{4.17} \\
          \end{align*}

          then for an exact differential:

          \begin{align*}
            \left(\frac{\delta M}{\delta Y}\right)_X &=
            \left(\frac{\delta N}{\delta X}\right)_Y \tag{4.25}
            \label{eq:cauchy-riemann}
          \end{align*}

          In this case:

          \begin{align*}
            dz &= \overbrace{y}^{M}dx - \overbrace{x}^{N}dy \\
            \left(\frac{\delta M}{\delta Y}\right)_X &= 1 \\
            \left(\frac{\delta N}{\delta X}\right)_Y &= 1 \\
            \Aboxed{1 &= 1}
          \end{align*}

          \textbf{Because \cref{eq:cauchy-riemann} is satisfied, this
          is an exact differential.}

        }

        \pagebreak

      \item $dz = -\frac{xy}{w^2}dw + \frac{y}{w} dx + \frac{x}{w}dy$

        \boxedanswersmall{

          This requires three conditions to be satisfied.

          \begin{align*}
            dz &= \overbrace{-\frac{xy}{w^2}}^Mdw +
            \overbrace{\frac{y}{w}}^N dx + \overbrace{\frac{x}{w}}^Pdy \\
            \left(\frac{\delta M}{\delta x}\right)_{w,y} &=
            \left(\frac{\delta N}{\delta w}\right)_{x,y} \text{
            (First condition)}\\
            \left(\frac{\delta M}{\delta x}\right)_{w,y} &=-\frac{y}{w^2} \\
            \left(\frac{\delta N}{\delta w}\right)_{x,y}
            &=-\frac{y}{w^2} \text{ (First condition satisfied)}\\
            \left(\frac{\delta M}{\delta y}\right)_{w,x} &=
            \left(\frac{\delta P}{\delta w}\right)_{x,y} \text{
            (Second condition)}\\
            \left(\frac{\delta M}{\delta y}\right)_{w,x} &= -\frac{x}{w^2} \\
            \left(\frac{\delta P}{\delta w}\right)_{w,x} &=
            -\frac{x}{w^2} \text{ (Second condition satisfied)}\\
            \left(\frac{\delta N}{\delta y}\right)_{w,x} &=
            \left(\frac{\delta P}{\delta x}\right)_{w,y} \text{
            (Third condition)}\\
            \left(\frac{\delta N}{\delta y}\right)_{w,x} &= \frac{1}{w} \\
            \left(\frac{\delta P}{\delta x}\right)_{w,y} &=
            \frac{1}{w} \text{ (Third condition satisifed)} \\
          \end{align*}

          \textbf{Because \cref{eq:cauchy-riemann} is satisfied, this
          is also an exact differential.}
        }

    \end{enumerate}

  \item {[15 pts]} Consider the following equation:
    \begin{equation*}
      z(t, u, v) = \frac{t^3 u^2}{1 - v}
    \end{equation*}
    \begin{enumerate}[(a)]
      \item If $m$ is conjugate to $t$, $n$ is conjugate to $u, p$ is
        conjugate to $v$, determine the coefficient relations.

        \boxedanswersmall{

          The provided conjugate pairs indicate that the differential
          form of $z$ is:

          \begin{align*}
            dz &= m dt + n du + pdv
          \end{align*}

          Therefore, the coefficient relations are:

          \begin{align*}
            m &= \left(\frac{\delta z}{\delta t}\right)_{u, v} \\
            n &= \left(\frac{\delta z}{\delta u}\right)_{t, v} \\
            p &= \left(\frac{\delta z}{\delta v}\right)_{t, u}
          \end{align*}
        }

      \item Calculate $\psi(m, u, p) = z - mt - pv$: the Legendre
        transform of $z$, exchanging $t$ and $v$ for $m$ and $p$.
        Make sure that $t$ and $v$ do not appear in $\psi(m, u, p)$.

        \boxedanswersmall{

          \begin{align*}
            m &= \frac{dz}{dt} = \frac{3t^2u^2}{1 - v} \\
            t &= \sqrt{\frac{m(1-v)}{3u^2}} \\
            p &= \frac{dz}{dv} = \frac{t^3u^2}{(1-v)^2} \\
            v &= 1 - \sqrt{\frac{t^3u^2}{p}} \\
            t &= \sqrt{\frac{m\left[1-\left( 1 -
            \sqrt{\frac{t^3u^2}{p}}\right)\right]}{3u^2}} \\
            1 - \frac{3u^2t^2}{m} &= \left( 1 -
            \sqrt{\frac{t^3u^2}{p}}\right) \\
            \frac{3u^2t^2}{m} &= \sqrt{\frac{t^3u^2}{p}} \\
            \frac{9u^4t^4}{m^2} &= \frac{t^3u^2}{p} \\
            \frac{9u^4t}{m^2} &= \frac{u^2}{p} \\
            t &= \frac{m^2}{9pu^2} \\
            v &= 1 - \sqrt{\frac{\left(\frac{m^2}{9pu^2}\right)^3u^2}{p}}
          \end{align*}
        }

        \boxedanswersmall{
          \begin{align*}
            (1 - v)^2 &= \frac{\left(\frac{m^2}{9pu^2}\right)^3u^2}{p} \\
            v &= 1 - \sqrt{\frac{m^6}{9^3p^4u^4}} \\
            \psi(m, u, p) &= z - mt - pv \\
            \Aboxed{\psi(m, u, p) &=
              \frac{\left(\frac{m^2}{9pu^2}\right)^3u^2}{
              - \sqrt{\frac{m^6}{9^3p^4u^4}}} -
              \left(\frac{3\left(\frac{m^2}{9pu^2}\right)^2u^2}{-
              \sqrt{\frac{m^6}{9^3p^4u^4}}}\right)\left(\frac{m^2}{9pu^2}\right)
              -
              \left(\frac{\left(\frac{m^2}{9pu^2}\right)^3u^2}{\left(-
              \sqrt{\frac{m^6}{9^3p^4u^4}}\right)^2}\right)\left(1
            - \sqrt{\frac{m^6}{9^3p^4u^4}}\right)}
            \\
          \end{align*}
        }

    \end{enumerate}

  \item {[25 pts]} A gas was determined to have an internal energy of
    the form $U = U_0 + 5PV$, where $U_0$ is a constant. Given the
    three states
    \begin{itemize}[\textbullet]
      \item state A: $P = \SI{0.2}{\mega\pascal},  V =
        \SI{0.01}{\meter\cubed}$
      \item state B: $P = \SI{0.2}{\mega\pascal}, V = \SI{0.03}{\meter\cubed}$
      \item state C: $P = \SI{0.5}{\mega\pascal}, V = \SI{0.01}{\meter\cubed}$
    \end{itemize}

    \boxedanswersmall{
      \centering
      \input{./assets/q_3_fig.tex}
      \vspace{0.5em}
      \captionof{figure}{PV diagram of three states with arrows
      indicating state changes.}
    }

    \begin{enumerate}[(a)]

        \pagebreak

      \item Calculate $Q$ and $W$ for the process taking state A to
        state B along a straight path

        \boxedanswersmall{
          \begin{align*}
            dU &= \overbrace{TdS}^{Q} - \overbrace{PdV}^{W} + \delta W' \\
            & \delta W' = 0 \\
            W &= PdV \\
            &
            \begin{aligned}
              P &= \SI{0.2}{\mega\pascal} \\
              dV &= \SI{0.03}{\meter\cubed} - \SI{0.01}{\meter\cubed} \\
              dV &= \SI{0.02}{\meter\cubed}
            \end{aligned} \\
            W &= (\SI{0.2e6}{\pascal})(\SI{0.02}{\meter\cubed}) \\
            \Aboxed{W &= \SI{4000}{\joule}} \\
            \Delta U &= Q + W \\
            Q &=\delta U - W \\
            &
            \begin{aligned}
              \Delta U &= U_2 - U_1 \\
              U &= U_0 + 5 PV \\
              &
              \begin{aligned}
                U_2 &= U_0 + 5
                (\SI{0.2}{\mega\pascal})(\SI{0.03}{\meter\cubed}) \\
                U_1 &= U_0 + 5
                (\SI{0.2}{\mega\pascal})(\SI{0.01}{\meter\cubed}) \\
              \end{aligned} \\
              \Delta U &=
              5(\SI{0.2e6}{\pascal})(\SI{0.03}{\meter\cubed} -
              \SI{0.01}{\meter\cubed}) \\
              \Delta U &=\SI{20000}{\joule} \\
            \end{aligned} \\
            Q &= (\SI{20000}{\joule}) - (\SI{4000}{\joule}) \\
            \Aboxed{Q &= \SI{16000}{\joule}}
          \end{align*}
        }

        \pagebreak
      \item Calcualte $Q$ and $W$ for the process taking state B to
        state C along a straight path

        \boxedanswer{
          \begin{align*}
            W &= \int_{V_1}^{V_2} P(V) dV \\
            &
            \begin{aligned}
              P - P_1 &= \frac{\Delta P}{\Delta V}(V - V_1) \\
              P &= \frac{\Delta P}{\Delta V}V - \frac{\Delta P}{\Delta
              V}V_1 + P_1 \\
              P &= \frac{\Delta P}{\Delta V}V - \frac{\Delta P}{\Delta
              V}V_1 + P_1 \\
            \end{aligned} \\
            W &= \int_{V_1}^{V_2}\frac{\Delta P}{\Delta V}V -
            \frac{\Delta P}{\Delta
            V}V_1 + P_1 dV \\
            W &= \frac{\Delta P}{\Delta
            V}\overbrace{\int_{V_1}^{V_2}VdV}^{\left[\frac{V^2}{2}\right]_{V_1}^{V_2}}
            -
            \frac{\Delta P\cdot V_1}{\Delta V}(V_2 - V_1) + P_1(V_2 - V_1) \\
            W &= \frac{\Delta P}{2\Delta V}(V_2^2 - V_1^2)
            - \frac{\Delta
            P\cdot V_1}{\Delta V}(V_2 - V_1) + P_1(V_2 - V_1) \\
            W &= \frac{\Delta P}{2\Delta V}(V_2^2 - V_1^2)
            - \left(\frac{\Delta
            P\cdot V_1}{\Delta V}+ P_1\right)(V_2 - V_1)  \\
            &
            \begin{aligned}
              \Delta P &= P_2 - P_1 = \SI{0.3e6}{\pascal} \\
              \Delta V &= V_2 - V_1 = \SI{0.02}{\meter\cubed} \\
              V_1 &= \SI{0.03}{\meter\cubed} \\
              V_2 &= \SI{0.01}{\meter\cubed} \\
              P_1 &= \SI{0.2e6}{\pascal} \\
              P_2 &= \SI{0.5e6}{\pascal} \\
            \end{aligned} \\
            W &=
            \frac{(\SI{0.3e6}{\pascal})}{2(\SI{0.02}{\meter\cubed})}[(\SI{0.01}{\meter\cubed})^2
            - (\SI{0.03}{\meter\cubed})^2] \\
            &\hspace{1 cm}- \left(\frac{(\SI{0.3e6}{\pascal})\cdot
              (\SI{0.03}{\meter\cubed})}{(\SI{0.02}{\meter\cubed})} +
            (\SI{0.2e6}{\pascal})\right)[(\SI{0.01}{\meter\cubed})
            - (\SI{0.03}{\meter\cubed})]  \\
            \Aboxed{W &= \SI{1000}{\joule}} \\
            Q &= \Delta U - W \\
            \Delta U &= 5(P_2V_2 - P_1V_1) \\
            \Delta U &=
            5[(\SI{0.5e6}{\pascal})(\SI{0.01}{\meter\cubed}) -
              (\SI{0.2e6}{\pascal})(\SI{0.03}{\meter\cubed})
            ] \\
            \Delta U &= \SI{-5000}{\joule} \\
            Q &= \SI{-5000}{\joule} -  \SI{1000}{\joule} \\
            \Aboxed{Q &= \SI{-6000}{\joule}}
          \end{align*}
        }

        \pagebreak

      \item Calculate $Q$ and $W$ for the process taking state C to
        state A along a straight path

        \boxedanswer{
          Because there is no change in volume, no work is being done.

          \begin{align*}
            W &= P\overbrace{dV}^{0} \\
            \Delta U &= Q + \cancel{W} \\
            Q &= \Delta U  = U_2 - U_1\\
            &
            \begin{aligned}
              \Delta U &= 5V(P_2 - P_1) \\
              \Delta U &=
              5(\SI{0.01}{\meter\cubed})(\SI{0.5}{\mega\pascal} -
              \SI{0.2}{\mega\pascal}) \\
              \Delta U &= \SI{15000}{\joule}
            \end{aligned} \\
            \Aboxed{Q &= \SI{15000}{\joule}} \\
            \Aboxed{W &= \SI{0}{\joule}}
          \end{align*}
        }

        \pagebreak

      \item Calculate $Q$ and $W$ for the process taking state A to
        state B along the parabolic path $P = P_0 +
        \kappa\left(\frac{V}{V_0} - 1\right)^2$ where
        $P_0 = \SI{0.1}{\mega\pascal}, V_0 = \SI{0.02}{\meter\cubed}$, and
        $\kappa = \SI{0.4}{\mega\pascal}$.

        \boxedanswersmall{
          \begin{align*}
            W &= \int_{V_1}^{V_2} P(V) dV \\
            W &= \int_{V_1}^{V_2}  P_0 +
            \kappa\left(\frac{V}{V_0} - 1\right)^2dV \\
            &
            \begin{aligned}
              u &= \frac{V}{V_0} - 1 \\
              du &= \frac{dV}{V_0} \\
              dV &= V_0du \\
            \end{aligned} \\
            W &=   P_0(V_2 - V_1) +
            \kappa V_0\int_{V_1}^{V_2} u^2du \\
            W &=   P_0(V_2 - V_1) +
            \kappa \frac{V_0}{3}\left[\left(\frac{V}{V_0} -
            1\right)^3\right]_{V_1}^{V_2} \\
            &
            \begin{aligned}
              P_0 &= P_1 = P_2 = \SI{0.1}{\mega\pascal} \\
              V_0 &= \SI{0.02}{\meter\cubed} \\
              V_1 &= \SI{0.01}{\meter\cubed} \\
              V_2 &= \SI{0.03}{\meter\cubed} \\
              \kappa &= \SI{0.4}{\mega\pascal} \\
            \end{aligned} \\
            W &=
            (\SI{0.1}{\mega\pascal})(\SI{0.03}{\meter\cubed}-
            \SI{0.01}{\meter\cubed}) + \\
            &\hspace{1
            cm}(\SI{0.4}{\mega\pascal})\frac{(\SI{0.02}{\meter\cubed})}{3}\left[\left(\frac{\SI{0.03}{\meter\cubed}}{(\SI{0.02}{\meter\cubed})}
                -
              1\right)^3 -
              \left(\frac{\SI{0.01}{\meter\cubed}}{(\SI{0.02}{\meter\cubed})}
                -
            1\right)^3\right] \\
            \Aboxed{W &= \SI{2666.6}{\joule}} \\
            Q &= \Delta U - W \\
            &
            \begin{aligned}
              \Delta U &= 5P_1(V_2 - V_1) \\
              \Delta U &=
              5(\SI{0.1e6}{\mega\pascal})[(\SI{0.03}{\meter\cubed}) -
              (\SI{0.01}{\meter\cubed})] \\
              \Delta U &= \SI{10000}{\joule} \\
              \Aboxed{Q &= \SI{7333.3}{\joule}}
            \end{aligned} \\
          \end{align*}
        }

    \end{enumerate}

  \item  {[35 pts]} Consider an isolated system (no heat, matter, or
    work may cross the system boundary)
    consisting of three internal compartments A, B, and C.
    Compartment A is 1 liter, B is 2 liters, and
    C is 4 liters. The compartments are separated by partitions; each
    partition has a valve (V1 and V2)
    which may be opened remotely. Initially the central volume (B) is
    filled with an ideal gas at 298K and
    \SI{354.6}{\kilo\pascal} and the other two (A and C) are
    completely evacuated (i.e., \SI{0}{\pascal}).

    \begin{figure}[ht]
      \centering
      \includegraphics[width=0.6\textwidth]{./assets/q_4_fig.png}
    \end{figure}

    \begin{enumerate}[(a)]
      \item Consider the following processes:

        \textit{First process: The valve to the A side is opened, the gas
          expands freely into compartment A, and
          the system comes to equilibrium. Then the valve to the C side
          is opened, and the system comes
        to equilibrium again.}

        \hspace{0.5 cm}

        \textit{Second process: Both valves are opened simultaneously, the
          gas expands freely into both compart-
        ments, and the A/B/C system comes to equilibrium.}

        \hspace{0.5 cm}

        Which of the processes above produces more entropy? You must
        explain your answer completely.

        \boxedanswer{
          In this system,
          because the system is isolated, the internal energy does not change.
          $\Delta U = Q + W = 0$. Hence, $Q = W$. The internal energy
          of an idea gas can be found:

          \begin{align*}
            \Delta U &= C_V(T_2 - T_1) = \frac{3}{2}R(T_2 - T_1) \tag{4.62} \\
          \end{align*}

          Thus the internal energy is only dependent on temperature
          for an ideal gas. Since the internal energy does not
          change, and an expansion is taking place, the final
          temperature is equal to the initial temperature in both processes.

          Thus, we can produce a general expression for the entropy
          change given this information.

          \begin{align*}
            dS &= \frac{C_V}{T}\overbrace{dT}^0 + \frac{\alpha}{\beta}dV \\
            dS &=  \frac{\alpha}{\beta}dV \\
            &
            \begin{aligned}
              \alpha &= \frac{1}{T} \\
              \beta &= \frac{1}{P} \\
            \end{aligned}\tag{4.57-4.58} \\
            dS &=  \frac{P}{T}dV \\
            & \frac{P}{T} = \frac{R}{V} \\
            dS &= \frac{R}{V}dV  \\
            \int_{S_1}^{S_2} dS &= R\int_{V_1}^{V_2} \frac{1}{V}dV  \\
            \Delta S &= R\ln\left(\frac{V_2}{V_1}\right)
            \tag{1}\label{eq:entropy} \\
          \end{align*}

        }

        \pagebreak

        \boxedanswersmall{
          Based on the above relation, we can compare the two processes:
          \begin{align*}
            \Delta S_1 &=   R\ln\left(\frac{V_2}{V_1}\right) +
            R\ln\left(\frac{V_3}{V_2}\right) \\
            \Delta S_1 &=
            R\ln\left(\frac{\cancel{V_2}V_3}{V_1\cancel{V_2}}\right) \\
            \Delta S_2 &=
            R\ln\left(\frac{V_3}{V_1}\right) \\
            % &
            % \begin{aligned}
            %   R &= \SI{8.314}{\joule\per\mole\kelvin} \\
            % V_1 &= \SI{1}{\liter} = \SI{1e-3}{m^3} \\
            % V_2 &= \SI{1}{\liter} + \SI{2}{\liter} = \SI{3e-3}{m^3} \\
            % V_3 &= \SI{1}{\liter} + \SI{2}{\liter} + \SI{4}{\liter}
            % = \SI{7e-3}{m^3} \\
            % \end{aligned} \\
            % \Delta S_1 &=
            % (\SI{8.314}{\joule\per\mole\kelvin})\ln\left(\frac{(\SI{3e-3}{\meter\cubed})(\SI{7e-3}{\meter\cubed})}{(\SI{1e-3}{\meter\cubed})(\SI{3e-3}{\meter\cubed})}\right)
            % \\
            % \Delta S_1 &= \SI{16.18}{\joule\per\kelvin\mole} \\
            % \Delta S_2 &=   R\ln\left(\frac{V_3}{V_1}\right) \\
            % \Delta S_2 &=
            % (\SI{8.314}{\joule\per\mole\kelvin})\ln\left(\frac{(\SI{7e-3}{\meter\cubed})}{(\SI{1e-3}{\meter\cubed})}\right)
            % \\
          \end{align*}

          \textbf{Based on the above, the total entropy change relies
            solely on the initial and final state of the system. Hence,
            both processes produces the same amount of entropy. This
            makes sense, as entropy is a state function, and the change
            in entropy relies only on the initial and final state of
            the system. Because the system is isolated, no entropy
            enters or leaves the system, so the amount generated in
          both processes must be the same.}

        }

      \item  The internal energy of an ideal gas is determined solely
        by its temperature. If that’s the case,
        what is the final temperature of the system in the first process?

        \boxedanswer{
          As explained above, the internal energy does not change,
          and the temperature remains the same in the first process.
        }

      \item It is found that the change in entropy of the first
        process is \SI{5.5}{\joule\per\kelvin}. What can you say about
        the reversibility of the process?

        \boxedanswer{
          \cref{eq:entropy} can be used to calculate the change in entropy.

          \begin{align*}
            \Delta S &= R\ln\left(\frac{V_3}{V_1}\right) \\
            &
            \begin{aligned}
              R &= \SI{8.314}{\joule\per\mole\kelvin} \\
              V_1 &= \SI{1}{\liter} = \SI{1e-3}{m^3} \\
              V_3 &= \SI{1}{\liter} + \SI{2}{\liter} + \SI{4}{\liter}
              = \SI{7e-3}{m^3} \\
            \end{aligned} \\
            \Delta S &=
            (\SI{8.314}{\joule\per\mole\kelvin})\ln\left(\frac{(\SI{7e-3}{m^3})}{(\SI{1e-3}{m^3})}\right)
            \\
            \Delta S &= \SI{16.18}{\joule\per\mole\kelvin}
          \end{align*}

          \textbf{\SI{5.5}{\joule\per\kelvin} is less than
            \SI{16.18}{\joule\per\mole\kelvin}, which means that the
            original constraints of the system must have been violated.
            Either the gas is not ideal, or some work/heat crossed the
            system boundary. Alternatively, the final volume may just
            be smaller than originally specified. Whether the process
            is reversible or not depends on the nature of the work
            extracted from the
            system. If no work was extracted, then the process is still
            irreversible. If some was, then it is at least partially
            reversible, possible fully reversible depending on the
            nature of the expansion. However, sudden expansions (like
          the one considered here) are almost entirely irreversible.}

        }

    \end{enumerate}

  \item {[15 pts]}
    \begin{enumerate}[(a)]
      \item If $\psi(m) = z − mx$ is the Legendre
        transform of $z(x)$ (where $m$ is conjugate to $x$), prove
        that
        \[\frac{d^2\psi}{dm^2} = -\left[\frac{d^2z}{dx^2} \right]^{-1}\]

        Hint: the chain rule will come in handy,
        as will the identity
        $dx/dz = 1/(dz/dx)$.

        \boxedanswer{
          \begin{align*}
            \psi(m) &= z(x) - mx \\
            m &= \frac{dz}{dx}  \\
            \frac{d\psi}{dm} &= \frac{dz}{dx}\frac{dx}{dm} -
            \left(m\frac{dx}{dm} + x\right) \\
            \frac{d\psi}{dm} &= (\overbrace{z'(x) - m}^{0})\frac{dx}{dm} - x \\
            \frac{d\psi}{dm} &= -x \\
            \frac{d^2\psi}{dm^2} &= -\frac{dx}{dm} \\
            &
            \begin{aligned}
              \frac{dx}{dm} &= \frac{1}{dm/dx} = \frac{1}{z''(x)} \\
            \end{aligned} \\
            \Aboxed{\frac{d^2\psi}{dm^2} &= -\left[ \frac{d^2
            z}{dx^2} \right]^{-1}}
          \end{align*}
        }

      \item Describe what this implies for the graph of $\psi$
        relative to the graph of $z$.

        \boxedanswer{
          \textbf{The concavity of the curve of the Legendre
          transformation is opposite that of $z$.}
        }

    \end{enumerate}

\end{enumerate}
