that, if some function $\psi(z)$ is a

  \item $z\frac{d\psi(z)}{dz} + g(z)\psi(z) = 0$ where $g(z)$ is some
    specific function

      \begin{align*}
        1 + 1 = 2
      \end{align*}
    }

  \item $\psi(z)\frac{d\psi(z)}{dz} + \psi(z) = 0$

    \boxedanswersmall{
      Some answer:
      \begin{align*}
        1 + 1 = 2
      \end{align*}
    }

\end{enumerate}
